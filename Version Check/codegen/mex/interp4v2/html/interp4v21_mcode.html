  4a-agrid

 13%}

 19% (1) Calculate the locations of a

 21na  = length(a);  % num of grid points
 22nav = length(av); % num of interpolation points
 23amax = na;

 27yyy =y;
 28%{
 29while amax>amin+1
 30
 31tempn = floor((amax+amin)/2);
 32
 33if av(1)>a(tempn)
 34    amin = tempn;
 35else
 36    amax = tempn;
 37end
 38 end
 39 
 40 avloc(1,:) = [amin amax];
 41 zd(1)      = (zv(1) - z(zind(1,1)))/(z(zind(1,2)) - z(zind(1,1)));
